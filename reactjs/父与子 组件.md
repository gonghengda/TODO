# 父子组件传递

``` 
  // 文件 ./AAA.js

  import React,{Component} from 'react';
  import PropTypes from 'prop-types';

  export default class AAA extends Component {
    getChildContext(){
      return {
        component:this,
        aaa: this.aaa.bind(this)
      }
    }


    aaa(){
      return '你好';
    }
    render{
      return(
         <div>666<div> 
      )
    }
  }

  MeetingDash.childContextTypes = {
    component: PropTypes.any,
    aaa: PropTypes.any,
  };
```
